# Resources

Pokemon
  - ID: number
  - Artwork: string (url)
  - Name: string
  - Types: PokemonType[]
  - Weight: number
  - Height: number
  - Abilities: PokemonAbility[]
  - Statistics: PokemonStat[]
  - Moves: Move[]

PokemonType
  - Name: string
  - Color: string (hex)

PokemonAbility
  - Name: string

PokemonStat
  - Name: string
  - Scale: number (from 1 to 100)

Move
  - Name: string
  - Accuracy: number
  - Power: number
  - PowerPoints: number
